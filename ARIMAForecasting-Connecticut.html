# Define the range of p, d, and q

d_value <- 2

# Iterate over all combinations of p, d, q
for (p in p_values) {
  for (q in q_values) {
    model <- Arima(ct_ts, order = c(p, d_value, q))
    model_name <- paste("ARIMA(", p, ",", d_value, ",", q, ")", sep = "")
    models[[model_name]] <- model
    bic_values[model_name] <- BIC(model)
  }
}
